# Структура проекта

```
┌── gulp                                      	# папка настроек и таск галпа
│   ├── config
│   │   ├── paths.js                            # глобальная переменная с путями к папкам и файлам
│   │   └── read-dir.js                         # функция возвращающая массив названий js файлов для вебпака
│   │
│   └── tasks
│       ├── fonts.js                            # конвертирование, обработка шрифтов
│       ├── html.js                             # обработка njk, конвертирование в html
│       ├── images.js                           # минимизация картинок, обработка svg
│       ├── reset.js                            # очищает build
│       ├── root.js                             # копирует содержимое папки './src/assets/root' в build и json в корень
│       ├── scripts.js                          # подключает, минимизирует библиотеки и файлы js
│       ├── serv.js                             # запускает локальный сервер
│       └── style.js                            # подключает, минимизирует css и scss файлы
│
├── readme                                      # readme
│   ├── check-list.md
│   └── ...
│
├── src                                         # папка исходников
│   ├── assets                                  # картинки, скрипты, попапы, стили, ...
│   │   ├── fonts                               # шрифты
│   │   │   ├── OpenSans-Bold.ttf
│   │   │   └── ...
│   │   │
│   │   ├── img                                 # картинки
│   │   │   ├── svg                             # иконки
│   │   │   │   └── arrow-right.svg
│   │   │   │
│   │   │   └── logo.svg
│   │   │
│   │   ├── js                                  # скрипты
│   │   │   ├── components                      # основная папка функций
│   │   │   │   ├── ajax-form.js                # отправка данных форм с помощью ajax
│   │   │   │   ├── check-inputs.js             # проверка input-ов на заполненность
│   │   │   │   ├── cookie-accept.js            # функция появления куки попапа
│   │   │   │   ├── cookie.js                   # функции куки
│   │   │   │   ├── custom-select.js            # заменяет стандартный select на настраиваемый
│   │   │   │   ├── header-mobile-menu.js       # переделывает меню в шапке в мобильное
│   │   │   │   ├── input-file.js               # добавляет новые возможности к input file
│   │   │   │   ├── map.js                      # яндекс карта
│   │   │   │   ├── options.js                  # основные данные браузера и глобальные переменные проекта
│   │   │   │   ├── phone-mask.js               # маска телефона для input-ов
│   │   │   │   ├── popups.js                   # делает ссылки на попапы
│   │   │   │   ├── sliders.js                  # инициализация swiper слайдеров с помощью дата атрибутов
│   │   │   │   └── validation-forms.js         # настраиваемая валидация форм
│   │   │   │
│   │   │   └── main.js                         # основной скрипт
│   │   │
│   │   ├── root                                # папка с дополнительными файлами (видео, аудио, документы, ...)
│   │   │   └── favicon.svg
│   │   │
│   │   └── style                               # стили
│   │       ├── components                      # основная папка с файлами стилей
│   │       │   ├── _btn.scss
│   │       │   └── ...
│   │       │
│   │       ├── modules                         # папка функциональных файлов
│   │       │   ├── _functions.scss             # функции
│   │       │   ├── _main.scss                  # файл включающий основные миксины и переменные
│   │       │   ├── _mixins.scss                # миксины
│   │       │   └── _variables.scss             # переменные
│   │       │
│   │       ├── main.scss                       # основной файл стилей
│   │       └── sitemap.scss                    # стили карты сайта
│   │
│   ├── popups                                  # попапы
│   │   ├── acceptcookie.njk                    # куки
│   │   └── postmessage.njk                     # сообщение после отправки формы
│   │
│   ├── templates                               # шаблоны
│   │   ├── blocks
│   │   │   ├── form.njk                        # шаблон формы html - если не нужно,- удалить
│   │   │   └── map.njk                         # шаблон карты - если не нужно,- удалить
│   │   │
│   │   ├── macros.njk                          # макросы
│   │   └── template.njk                        # основной html шаблон
│   │
│   ├── 404.njk                                 # страница 404 ошибки
│   ├── customization.json                      # json файл настроек вида для карты
│   ├── index.njk                               # главная страница
│   ├── response-form.json                      # имитация ответа сервера после отправки формы
│   └── sitemap.njk                             # страница карты сайта
│
├── .editorconfig                               # настройки единого формата для всех языков и IDE
├── .gitignore                                  # перечень игнорируемых файлов, папок для отправки в git
├── .gitlab-ci.yml                              # настройки ci\cd в gitlab
├── gulpfile.js                                 # файл конфигурации gulp
├── package-lock.json                           # файл в котором указываются все установленные пакеты и их зависимости
├── package.json                                # метаданные, зависимости, библиотеки проекта
└── webpack.config.js                           # webpack настройки для подключения библиотек из node_modules
